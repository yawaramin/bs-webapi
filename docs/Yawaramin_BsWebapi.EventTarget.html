Module Yawaramin_BsWebapi.EventTarget

module EventTarget: Yawaramin_BsWebapi_EventTarget
type 'a t = 'a Yawaramin_BsWebapi_Common.t 
type intf = [ `eventTarget ] 
type options = < capture : bool; once : bool; passive : bool > Js.t 
val make : unit -> intf t
val addEventListener : type_:string ->
       listener:(unit -> unit) ->
       ?options:options -> unit
addEventListener ~type_ ~listener ?options t adds an event handler listener, of type type_, with options options, to the event target t.

listener must be a BuckleScript uncurried function, e.g. fun [@bs] () -> Js.log "Hello, World!".
